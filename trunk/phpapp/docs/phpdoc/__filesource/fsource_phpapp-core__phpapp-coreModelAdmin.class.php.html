  437.         public function updateModel($model, $params, $flags = 0, $preloadRelations = false) {
  438.  
  439.             $m2m = array();
  440.  
  441.             // Enforce access
  442.             $modelAccess = $this->getModelAccess();
  443.             $master = $this->module->master;
  444.             $requiredFilter = $this->module->requiredFilter;
  445.             $filteredMaster = false;
  446.  
  447.             foreach ($model->_fields as $field => $def) {
  448.                 if ($access = $this->getMatchedFields($field)) {
  449.                     $path = explode('.', $access);
  450.                     $relation = $this->getUser();
  451.                     foreach ($path as $rel)
  452.                         $relation = $relation->$rel;
  453.                     if ($relation) {
  454.                         if (!$id)
  455.                             $model->$field = $relation->pk;
  456.                         $def->options['editable'] = false;
  457.                     }
  458.                 }
  459.  
  460.                 if (isset($this->fieldRestrictions[$field]) && $restricted = $this->fieldRestrictions[$field]) {
  461.                     if (!$this->checkPermission($restricted[1])) { 
  462.                         unset($params[$field]);
  463.                         $def->options['editable'] = false;
  464.                     }
  465.                 } elseif (isset($modelAccess[$field])) {
  466.                     $params[$field] = $modelAccess[$field];
  467.                     $def->options['editable'] = false;
  468.                 }
  469.  
  470.                 if ($master != $this) {
  471.                     if ($def instanceof RelationField && $master && $def->relationName == $master->getQueryName()) {
  472.                         $filteredMaster = true;
  473.                         if ($filterid = $this->getAdmin()->getFilter($master->getQueryName())) {
  474.                             if ($def instanceof ManyToOneField) {
  475.                                 $params[$field] = $filterid;
  476.                                 $def->options['editable'] = false;
  477.                             } elseif ($def instanceof ManyToManyField) {
  478.                                 $rq = $this->getAdmin()->findModelAdmin($def->relationName)->getQuery();
  479.                                 $ro = $rq->get($filterid);
  480.                                 $m2m[$field] = $ro;
  481.                             }
  482.                         }
  483.                     }
  484.                 }
  485.  
  486.                 if (!$filteredMaster && $requiredFilter) {
  487.                     if ($def instanceof ManyToOneField && $def->relationName == $requiredFilter) {
  488.                         if ($filterid = $this->getAdmin()->getFilter($requiredFilter)) {
  489.                             $params[$field] = $filterid;
  490.                             $def->options['editable'] = false;
  491.                         }
  492.                     }
  493.                 }
  494.  
  495.                 // Set default value for base filter fields
  496.                 if ($baseFilter = $this->getAdmin()->getBaseFilter()) {
  497.                     if ($def instanceof ManyToOneField && $def->relationName == $baseFilter->getQueryName()) {
  498.                         if ($filterid = $this->getAdmin()->getFilter($baseFilter->getQueryName())) {
  499.                             $params[$field] = $filterid;
  500.                         }
  501.                     }
  502.                 }
  503.  
  504.                 if (count($this->queryFilter)) {
  505.                     foreach ($this->queryFilter as $f => $v)
  506.                         if ($f == $field) {
  507.                             $params[$f] = $v;
  508.                             $def->options['editable'] = false;
  509.                         }
  510.                 }
  511.  
  512.             }
  513.  
  514.             // Prevent boolean fields from getting reset if they aren't
  515.             // in the current form view
  516.             if ($flags & UPDATE_FORCE_BOOLEAN) {
  517.                 $displayedFields = $this->getFlattenedFieldGroups();
  518.                 foreach ($model->_fields as $n => $f)
  519.                     if ($f instanceof BooleanField)
  520.                         if (!in_array($n, $displayedFields)) {
  521.                             if ($model->pk)
  522.                                 $params[$n] = $model->$n;
  523.                             else
  524.                                 $params[$n] = $f->getDefaultValue();
  525.                         }
  526.             }
  527.  
  528.             $model->updateModel($params, null, $flags);
  529.  
  530.             if (count($m2m) && $preloadRelations) {
  531.                 // Pre-add many-to-many fields that are active filters
  532.                 if (!$model->pk)
  533.                     foreach ($m2m as $f => $v)
  534.                         $model->$f->preload(array($v));
  535.             }
  536.  
  537.         }
