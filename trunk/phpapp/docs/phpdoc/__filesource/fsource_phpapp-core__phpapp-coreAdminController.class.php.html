  228.         public function execute($path, $params, $method = 'get') {
  229.             $start = microtime(true);
  230.             $this->app->query->begin();
  231.             $path = explode('/', substr($path, strlen($this->baseUrl)+1));
  232.             if (!$path[sizeof($path)-1]) array_pop($path);
  233.  
  234.             if (!$this->app->isLoggedIn()) {
  235.                 if (($path[0] != 'account' || $path[1] != 'login') && $path[0] != 'media') {
  236.                     $this->relativeRedirect('/account/login?redirect='.rawurlencode($_SERVER['REQUEST_URI']));
  237.                     return;
  238.                 }
  239.             }
  240.  
  241.             foreach ($params as $k => $v)
  242.                 if (strpos($k, '__') !== false) {
  243.                     $params[str_replace('__', '.', $k)] = $v;
  244.                     unset($params[$k]);
  245.                 }
  246.  
  247.             try {
  248.                 $handler = array_shift($path);
  249.                 if ($handler == 'account') {
  250.                     header('Content-Type: text/html; charset=UTF-8');
  251.                     return $this->doAccountAction($path, $params, $method);
  252.                 } elseif ($handler == 'media') {
  253.                     return $this->sendMedia($path, $params, $method);
  254.                 } elseif ($handler == 'thumb') {
  255.                     return $this->sendThumbnail($path, $params, $method);
  256.                 } elseif (!$handler || $handler == 'modules') {
  257.                     header('Content-Type: text/html; charset=UTF-8');
  258.                     if (sizeof($path) == 0) {
  259.                         // Module index
  260.                         $this->renderTemplate('module_list.tpl', array('modules' => $this->modules));
  261.                     } else {
  262.                         $moduleid = array_shift($path);
  263.                         $module = $this->modules[$moduleid];
  264.                         if ($module)
  265.                             $module->execute($path, $params, $method);
  266.                         else
  267.                             throw new NotFoundException('The requested page does not exist.');
  268.                     }
  269.                 } else {
  270.                     throw new NotFoundException('The requested page does not exist.');
  271.                 }
  272.             } catch (NotFoundException $nfe) {
  273.                 $context = array('modules' => $this->modules,
  274.                         'title' => 'Page Not Found',
  275.                         'error' => $nfe->getMessage());
  276.                 $this->renderTemplate('error.tpl', $context);
  277.             } catch (AccessDeniedException $ade) {
  278.                 $context = array_merge(array('modules' => $this->modules,
  279.                         'title' => 'Access Denied',
  280.                         'error' => $ade->getMessage()),
  281.                     $ade->getContext());

  285.                         'title' => 'Error',
  286.                         'error' => $e->getMessage());

  290.         }

  316.                 case 'logout':
  317.                     $this->app->logout();
  318.                     $this->relativeRedirect('/account/login');
  319.                     break;
  320.                 case 'heartbeat':

  322.                     echo "\x47\x49\x46\x38\x39\x61\x01\x00\x01\x00\x80\x00\x00\xff\xff\xff\x00\x00\x00\x21\xf9\x04\x00\x00\x00\x00\x00\x2c\x00\x00\x00\x00\x01\x00\x01\x00\x00\x02\x02\x44\x01\x00\x3b";

  324.                 case 'save':

  326.                     $pass1 = $params['password1'];

  338.                     $user['first_name'] = $params['first_name'];
  339.                     $user['last_name'] = $params['last_name'];
  340.                     try {
  341.                         $user->save();
  342.                         $this->relativeRedirect('/account/saved');
  343.                     } catch (ValidationException $ve) {
  344.                         $context = array('errors' => $user->getErrors(),
  345.                             'fieldErrors' => $user->getFieldErrors());

  349.                 case 'saved':

  356.                 default:

  358.             }

  361.         public function sendMedia($path, $params, $method = 'get') {
  362.             $overrides = array(
  363.                 '.css' => 'text/css',
  364.                 '.html' => 'text/html',
  365.                 '.js' => 'text/javascript',
  366.                 '.css' => 'text/css',
  367.                 '.png' => 'image/png',
  368.                 '.jpg' => 'image/jpeg',
  369.                 '.jpeg' => 'image/jpeg',
  370.                 '.gif' => 'image/gif'
  371.             );
  372.             $relpath = '/'.implode('/', $path);
  373.             $mediapath = dirname(__FILE__).'/media'.$relpath;
  374.             if (file_exists($mediapath) && is_file($mediapath)) {
  375.             
  376.                 if (array_key_exists('HTTP_IF_MODIFIED_SINCE', $_SERVER)) {
  377.                     $lastmod = filemtime($mediapath);
  378.                     $lastreq = strtotime($_SERVER['HTTP_IF_MODIFIED_SINCE']);
  379.                     if (array_key_exists('HTTP_CACHE_CONTROL', $_SERVER)) {
  380.                         $cc = $_SERVER['HTTP_CACHE_CONTROL'];
  381.                         // max-age just specifies to validate against modified time
  382.                         //if ($cc == 'max-age=0' || $cc = 'no-cache')
  383.                         if ($cc == 'no-cache')
  384.                             $lastreq = 0;
  385.                     }
  386.                     if ($lastmod <= $lastreq) {
  387.                         header('HTTP/1.1 304 Not Modified');
  388.                         header('Pragma: phpsucks');
  389.                         header('Cache-Control: public');
  390.                         exit;
  391.                     }
  392.                 }
  393.  
  394.                 $ext = substr($mediapath, strrpos($mediapath, '.'));
  395.                 if ($ext == '.php') {
  396.                     include($mediapath);
  397.                 } else {
  398.                     if (array_key_exists($ext, $overrides))
  399.                         $mimetype = $overrides[$ext];
  400.                     else
  401.                         //$mimetype = mime_content_type($mediapath);
  402.                         $mimetype = 'text/plain';
  403.                     header('Pragma: phpsucks');
  404.                     header('Cache-Control: public');
  405.                     header('Content-type: '.$mimetype);
  406.                     header('Last-Modified: '.gmdate('D, d M Y H:i:s', filemtime($mediapath)).' GMT');
  407.                     if ($fh = fopen($mediapath, 'r')) {
  408.                         while (!feof($fh))
  409.                             echo fread($fh, 8192);
  410.                         fclose($fh);
  411.                     }
  412.                 }
  413.             } else {
  414.                 throw new NotFoundException();
  415.             }
  416.         }

  436.         private function getThumbnailPath($url) {
  437.             $cachedir = $this->app->getCachePath('thumbnails', true);
  438.             $thumbfile = $cachedir.'/thumb-'.md5($url);
  439.             if (!file_exists($thumbfile) || (time() - filemtime($thumbfile) > 60)) {
  440.                 $cachefile = $cachedir.'/cache-'.md5($url);
  441.                 file_put_contents($cachefile, file_get_contents($url));
  442.                 $imginfo = getimagesize($cachefile);
  443.                 $mime = $imginfo['mime'];
  444.                 $image = null;
  445.                 switch($mime) {
  446.                     case 'image/jpeg':
  447.                     case 'image/pjpeg':
  448.                         $image = imagecreatefromjpeg($cachefile);
  449.                         break;
  450.                     case 'image/png':
  451.                     case 'image/x-png':
  452.                         $image = imagecreatefrompng($cachefile);
  453.                         imagealphablending($image, false);
  454.                         imagesavealpha($image, true);
  455.                         break;
  456.                     case 'image/gif':
  457.                         $image = imagecreatefromgif($cachefile);
  458.                         break;
  459.                     default:
  460.                         return $cachefile;
  461.                 }
  462.                 if ($image) {
  463.                     $this->resizeImage($image, 100, 100, $thumbfile, 80, $mime);
  464.                     imagedestroy($image);
  465.                 }
  466.             }
  467.             return file_exists($thumbfile) ? $thumbfile : null;
  468.         }

  470.         private function resizeImage($im, $maxwidth, $maxheight, $urlandname, $comp, $imagetype) {
  471.             $width = imagesx($im);
  472.             $height = imagesy($im);
  473.             if(($maxwidth && $width > $maxwidth) || ($maxheight && $height > $maxheight)) {
  474.                 if($maxwidth && $width > $maxwidth) {
  475.                     $widthratio = $maxwidth/$width;
  476.                     $resizewidth = true;
  477.                 } else
  478.                     $resizewidth=false;
  479.  
  480.                 if($maxheight && $height > $maxheight) {
  481.                     $heightratio = $maxheight/$height;
  482.                     $resizeheight=true;
  483.                 } else
  484.                     $resizeheight=false;
  485.  
  486.                 if($resizewidth && $resizeheight) {
  487.                     if($widthratio < $heightratio) $ratio = $widthratio;
  488.                     else $ratio = $heightratio;
  489.                 } elseif($resizewidth) {
  490.                     $ratio = $widthratio;
  491.                 } elseif($resizeheight) {
  492.                     $ratio = $heightratio;
  493.                 }
  494.  
  495.                 $newwidth = $width * $ratio;
  496.                 $newheight = $height * $ratio;
  497.  
  498.                 if(function_exists('imagecopyresampled') && $imagetype !='image/gif')
  499.                     $newim = imagecreatetruecolor($newwidth, $newheight);
  500.                 else
  501.                     $newim = imagecreate($newwidth, $newheight);
  502.  
  503.                 // additional processing for png / gif transparencies (credit to Dirk Bohl)
  504.                 if($imagetype == 'image/x-png' || $imagetype == 'image/png') {
  505.                     imagealphablending($newim, false);
  506.                     imagesavealpha($newim, true);
  507.                 } elseif($imagetype == 'image/gif') {
  508.                     $originaltransparentcolor = imagecolortransparent( $im );
  509.                     if($originaltransparentcolor >= 0 && $originaltransparentcolor < imagecolorstotal($im)) {
  510.                         $transparentcolor = imagecolorsforindex( $im, $originaltransparentcolor );
  511.                         $newtransparentcolor = imagecolorallocate($newim,$transparentcolor['red'],$transparentcolor['green'],$transparentcolor['blue']);
  512.                         imagefill( $newim, 0, 0, $newtransparentcolor );
  513.                         imagecolortransparent( $newim, $newtransparentcolor );
  514.                     }
  515.                 }
  516.  
  517.                 imagecopyresampled($newim, $im, 0, 0, 0, 0, $newwidth, $newheight, $width, $height);
  518.  
  519.                 if($imagetype == 'image/pjpeg' || $imagetype == 'image/jpeg') {
  520.                     imagejpeg ($newim,$urlandname,$comp);
  521.                 } elseif($imagetype == 'image/x-png' || $imagetype == 'image/png') {
  522.                     imagepng ($newim,$urlandname,substr($comp,0,1));
  523.                 } elseif($imagetype == 'image/gif') {
  524.                     imagegif ($newim,$urlandname);
  525.                 }
  526.                 imagedestroy ($newim);
  527.             } else {
  528.                 if($imagetype == 'image/pjpeg' || $imagetype == 'image/jpeg') {
  529.                     imagejpeg ($im,$urlandname,$comp);
  530.                 } elseif($imagetype == 'image/x-png' || $imagetype == 'image/png') {
  531.                     imagepng ($im,$urlandname,substr($comp,0,1));
  532.                 } elseif($imagetype == 'image/gif') {
  533.                     imagegif ($im,$urlandname);
  534.                 }
  535.             }
  536.         }
