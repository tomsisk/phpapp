  377.         public function checkMultiPermission($action = null, $type = null, $module = null, $instance = null) {
  378.  
  379.             $action = $action ? strtoupper($action) : 'ALL';
  380.             $type = $type ? strtoupper($type) : 'ALL';

  383.  
  384.             $types = array($type);
  385.             $typemod = '&';
  386.  
  387.             if (strpos($type, '|') !== false) {
  388.                 $types = explode('|', $type);
  389.                 $typemod = '|';
  390.             } elseif (strpos($type, '&') !== false) {
  391.                 $types = explode('&', $type);
  392.                 $typemod = '&';
  393.             }
  394.  
  395.             $result = ($typemod == '|' ? false : true);
  396.  
  397.             foreach ($types as $t) {
  398.                 $key = $action.':'.$t.':'.$instance;
  399.                 $allowed = false;
  400.                 if (array_key_exists($key, $this->permCache)) {
  401.                     $allowed = $this->permCache[$key];
  402.                 } else {
  403.                     $allowed = $this->checkPermission($action, $module, $t, $instance);
  404.                     $this->permCache[$key] = $allowed;
  405.                 }
  406.  
  407.                 if ($typemod == '|' && $allowed) {
  408.                     $result = true;
  409.                     break;
  410.                 } elseif ($typemod == '&' && !$allowed) {
  411.                     $result = false;
  412.                     break;
  413.                 }
  414.             }
  415.  
  416.             return $result;
