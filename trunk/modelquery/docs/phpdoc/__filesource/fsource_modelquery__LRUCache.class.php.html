  221.         public function remove($data) {
  222.             $node = $this->_firstNode;
  223.             while ($node !== null) {
  224.                 if ($node->data == $data) {
  225.                     if ($node === $this->_firstNode)

  229.                     else {

  231.                         $node->next->previous = $node->previous;
  232.                         $this->_count--;
  233.                         return $node->data;
  234.                     }

  238.             return null;
  239.         }
