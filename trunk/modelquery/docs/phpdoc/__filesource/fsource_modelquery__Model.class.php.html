  51.     /**
  52.      * Data model representing a record in the database.
  53.      *
  54.      * Model is the base class for all of your data model definitions.
  55.      * To define a model, you must subclass this Model class.  Field
  56.      * definitions and configuration options should be handled by
  57.      * overriding Model::configure() in your subclass. When a QueryHandler
  58.      * handler is created for a Model subclass (by QueryFactory->get(),
  59.      * the model class is instantiated.  As part of the initialization,
  60.      * configure() is called to setup field definitions.
  61.      *
  62.      * Result sets returned by QueryFilter will be objects of the
  63.      * specified subclass type (except in the case of QueryFilter::hash()
  64.      * and QueryFilter::raw()).
  65.      *
  66.      * Subclass example:
  67.      *
  68.      * <code>
  69.      * class Book extends Model {
  70.      * 
  71.      *     public function configure() {
  72.      *         $this->id = new IntegerField('ID', array('pk' => true));
  73.      *         $this->title = new CharField('Title', 100, array('required' => true));
  74.      *         $this->published = new DateField('Publication Date', array(
  75.      *             'default' => CURRENT_TIMESTAMP));
  76.      *         $this->format = new CharField('Format', 20, array(
  77.      *             'options' => array('H' => 'Hardcover', 'P' => 'Paperback', 'A' => 'Audiobook')
  78.      *             ));
  79.      *         $this->setDefaultOrder('+title');
  80.      *     }
  81.      * 
  82.      *     public function __toString() {
  83.      *         return strval($this->title);
  84.      *     }
  85.      * 
  86.      * }
  87.      * </code>

  89.      * Implementation note: public attributes should not be accessed directly.

  91.      * by your code.

  237.         public function setFieldValue($name, $value, $raw = false) {
  238.             if ($name == 'pk')
  239.                 $name = $this->_idField;
  240.             $field = $this->_fields[$name];
  241.             if ($field) {
  242.                 if ($field instanceof RelationField) {
  243.                     $relid = $field->setRelation($this, $value);
  244.                     if ($value instanceof Model)
  245.                         $this->_fieldRelationValues[$name] = $value;
  246.                     if ($relid)
  247.                         $this->setPrimitiveFieldValue($name, $relid);
  248.                     else
  249.                         $this->setPrimitiveFieldValue($name, null);
  250.                 } else {
  251.                     $this->setPrimitiveFieldValue($name, $field->convertToDbValue($value), $raw);
  252.                 }
  253.             }
  254.         }

  266.                 $name = $this->_idField;
  267.             $this->_rawValues[$name] = $value;
  268.             if (isset($this->_fieldRelationValues[$name]))
  269.                 unset($this->_fieldRelationValues[$name]);
  270.             if ($raw)
  271.                 $this->_fieldValues[$name] = $value;
  272.             elseif ($this->_valuesConverted || array_key_exists($name, $this->_fieldValues)) {
  273.                 $field = $this->_fields[$name];
  274.                 $this->_fieldValues[$name] =
  275.                     $field->convertFromDbValue($this->_rawValues[$name]);
  276.             }
  277.             if (!array_key_exists($name, $this->_versionValues) || $this->_versionValues[$name] !== $value)
  278.                 $this->_versionChanges++;
  279.         }

  342.          * (i.e. $model->title), or via array access ($model['title']) since this

  351.         public function getFieldValue($name) {
  352.             if ($name == 'pk')
  353.                 $name = $this->_idField;
  354.             $field = $this->_fields[$name];
  355.             if ($field) {
  356.                 if ($field instanceof RelationField) {
  357.                     if (!array_key_exists($name, $this->_fieldRelationValues))
  358.                         try {
  359.                             $this->_fieldRelationValues[$name] = $field->getRelation($this, $this->getPrimitiveFieldValue($name));
  360.                         } catch (DoesNotExistException $dne) {
  361.                             return null;
  362.                         }
  363.                     return $this->_fieldRelationValues[$name];
  364.                 } else {
  365.                     return $this->getPrimitiveFieldValue($name);
  366.                 }
  367.             } else if (isset($this->_extraValues[$name])) {
  368.                 return $this->_extraValues[$name];
  369.             }
  370.             return null;
  371.         }

  386.                 return $this->_fieldValues[$name];

  388.                 $field = $this->_fields[$name];
  389.                 $this->_fieldValues[$name] =
  390.                     $field->convertFromDbValue($this->_rawValues[$name]);
  391.                 return $this->_fieldValues[$name];
  392.             }
  393.             return null;
  394.         }

  431.         /**
  432.          * Check if a field has been changed since the last save.
  433.          * @param string $name The field name, or null to check if <i>any</i>
  434.          *          fields in this model have changed.
  435.          * @return bool TRUE if the field or model has changed
  436.          */
  437.         public function isChanged($name = null) {
  438.             if ($name) {
  439.                 if ($name == 'pk')
  440.                     $name = $this->_idField;
  441.                 if (array_key_exists($name, $this->_versionValues))
  442.                     return ($this->_versionValues[$name] !== $this->_rawValues[$name]);
  443.                 else
  444.                     return array_key_exists($name, $this->_rawValues);
  445.             } else
  446.                 return ($this->_versionChanges > 0);
  447.         }
  448.  
  449.         /**
  450.          * Revert a field to the last saved value.
  451.          * @param string $name The field name, or null to revert <i>all</i>
  452.          *          fields in this model to the last saved value.
  453.          */
  454.         public function revert($name = null) {
  455.             if ($name) {
  456.                 if ($name == 'pk')
  457.                     $name = $this->_idField;
  458.                 if (array_key_exists($name, $this->_versionValues)) {
  459.                     $this->_rawValues[$name] = $this->_versionValues[$name];
  460.                     unset($this->_fieldValues[$name]);
  461.                     $this->_versionChanges--;
  462.                 } else if (array_key_exists($name, $this->_rawValues)) {
  463.                     unset($this->_rawValues[$name]);
  464.                     $this->_versionChanges--;
  465.                 }
  466.             } else {
  467.                 $this->_rawValues = $this->versionValues;
  468.                 $this->resetChangeTracking();
  469.             }
  470.         }
  471.  
  472.         /**
  473.          * Set a field value.
  474.          *
  475.          *If a field with the given property name exists, it will set it
  476.          * to a new value, internally calling Model::setFieldValue().
  477.          *
  478.          * When in configuration more (inside Model:configure()), it defines
  479.          * a field definition instead, internally calling Model::addField().
  480.          *
  481.          * @param string $name The field name
  482.          * @param mixed $value The field value
  483.          */
  484.         public function __set($name, $value) {
  485.             if ($name == 'pk')
  486.                 $name = $this->_idField;
  487.             if ($this->_configMode)
  488.                 $this->addField($name, $value);
  489.             else
  490.                 $this->setFieldValue($name, $value);
  491.         }
  492.  
  493.         /**
  494.          * Get a field value.
  495.          *
  496.          * If a field with the given property name exists, it will return the
  497.          * field's value, internally calling Model::getFieldValue().
  498.          *
  499.          * When in configuration more (inside Model:configure()), it returns
  500.          * the field definition instead, internally calling Model::getField().
  501.          *
  502.          * @param string $name The field name
  503.          * @return mixed The field value
  504.          */
  505.         public function __get($name) {
  506.  
  507.             if ($name{0} == '_')
  508.                 return $this->$name;
  509.             elseif ($name == 'pk')
  510.                 $name = $this->_idField;
  511.  
  512.             if ($this->_configMode)
  513.                 return $this->_fields[$name];
  514.             else
  515.                 return $this->getFieldValue($name);
  516.  
  517.         }
  518.  
  519.         /**
  520.          * Validate the model's current field data.
  521.          *
  522.          * If validation fails, errors can be retrieved from Model::getErrors()
  523.          * and Model::getFieldErrors().
  524.          *
  525.          * @param string $field The field to validate, or null for all fields
  526.          * @return bool TRUE if validation succeeds
  527.          */
  528.         public function validate($field = null) {
  529.  
  530.             if ($field)
  531.                 return $this->validateField($field);
  532.  
  533.             $valid = sizeof($this->_errors) == 0 && sizeof($this->_fieldErrors) == 0;
  534.             // Check for primary key changes
  535.             if ($this->_dbId && $this->pk && $this->_dbId != $this->pk) {
  536.                 if ($this->_query->filter('pk', $this->pk)->count() > 0) {
  537.                     $this->addFieldError($this->_idField, 'That '.strtolower($this->_fields[$this->_idField]->getName()).' is not available.');
  538.                     $valid = false;
  539.                 }
  540.             }
  541.             // Individual field validations
  542.             foreach ($this->_fields as $field => $def) {
  543.                 if (($field != $this->_idField || array_key_exists($this->_idField, $this->getFieldValues())) && !$def->validate($this->getFieldValue($field), $this)) {
  544.                     $errors = $def->validationErrors();
  545.                     if ($errors && count($errors))
  546.                         $this->addFieldErrors($field, $def->validationErrors());
  547.                     $valid = false;
  548.                 }
  549.             }
  550.             // Full object validations (multi-field dependencies)
  551.             foreach ($this->_validators as $validator) {
  552.                 if (!$validator[0]->validate($this)) {
  553.                     $this->addError($validator[1]);
  554.                     $valid = false;
  555.                 }
  556.             }
  557.             return $valid;
  558.         }
  559.  
  560.         /**
  561.          * Validate an individual field
  562.          */
  563.         private function validateField($field) {
  564.  
  565.             $valid = true;
  566.  
  567.             if (isset($this->_fields[$field])) {
  568.                 $def = $this->_fields[$field];
  569.                 unset($this->_fieldErrors[$field]);
  570.                 if (($field != $this->_idField || array_key_exists($this->_idField, $this->getFieldValues())) && !$def->validate($this->getFieldValue($field), $this)) {
  571.                     $errors = $def->validationErrors();
  572.                     if ($errors && count($errors))
  573.                         $this->addFieldErrors($field, $def->validationErrors());
  574.                     $valid = false;
  575.                 }
  576.             }
  577.  
  578.             return $valid;
  579.  
  580.         }
  581.  
  582.         /**
  583.          * Check if this model has been persisted to the data store.
  584.          *
  585.          * This does not verify that the current data has been persisted, just
  586.          * that the object originally was loaded from the data store.  To check
  587.          * if the current field data has been persisted, use Model::isChanged().
  588.          *
  589.          * @return bool TRUE if the object exists in the data store
  590.          * @see Model::isChanged()
  591.          */
  592.         public function isPersistent() {
  593.             return $this->_dbId != null;
  594.         }
  595.  
  596.         /**
  597.          * Set the database table name associated with this model.
  598.          * @param string $name_ The table name
  599.          */
  600.         public function setTable($name_) {
  601.             $this->_table = $name_;
  602.         }
  603.  
  604.         /**
  605.          * Get the database table name associated with this model.
  606.          * @return string The table name
  607.          */
  608.         public function getTable() {
  609.             return $this->_table;
  610.         }
  611.  
  612.         /**
  613.          * Set the default sorting order for queries on this model.
  614.          *
  615.          * This function takes an arbitrary number of arguments, each of
  616.          * which is a string containing an (optional) order specifier
  617.          * of <i>+</i> or <i>-</i> and a field name.

  621.         public function setDefaultOrder() {
  622.             $args = func_get_args();
  623.             $this->_defaultOrder = $args;
  624.         }
  625.  
  626.         /**
  627.          * Get the default sorting fields.
  628.          * @return Array A list of fields to order queries by
  629.          * @see Model::setDefaultOrder()
  630.          */
  631.         public function getDefaultOrder() {
  632.             return $this->_defaultOrder;
  633.         }
  634.  
  635.         /**
  636.          * Add a data validator to this module.
  637.          *
  638.          * In contrast to field validators, model validators are used
  639.          * to validate the model as a whole.  This is useful when one field's
  640.          * validation depends on the value of another field in the model.
  641.          * 
  642.          * @param ModelValidator $validator The validator instance
  643.          * @param string $message The error message to display if validation fails
  644.          * @see ModelValidator
  645.          * @see ModelField::addValidator()
  646.          * @see FieldValidator
  647.          */
  648.         public function addValidator($validator, $message) {
  649.             if (!$this->_validators) $this->_validators = array();
  650.             $this->_validators[] = array($validator, $message);
  651.         }
  652.  
  653.         /**
  654.          * Check if the last validation produced any field validation errors.
  655.          * @return bool TRUE if any field validators failed.
  656.          */
  657.         public function hasFieldErrors() {
  658.             return $this->_fieldErrors && sizeof($this->_fieldErrors) > 0;
  659.         }
  660.  
  661.         /**
  662.          * Clear any field validation errors from the last validation attempt.
  663.          */
  664.         public function clearFieldErrors() {
  665.             $this->_fieldErrors = array();
  666.         }
  667.  
  668.         /**
  669.          * Add a new field validation error.
  670.          * @param string $field The field name
  671.          * @param string $error The error message
  672.          */
  673.         public function addFieldError($field, $error) {
  674.             $this->addFieldErrors($field, array($error));
  675.         }
  676.  
  677.         /**
  678.          * Add a set of field validation errors.
  679.          * @param string $field The field name
  680.          * @param Array $error A list of error message
  681.          */
  682.         public function addFieldErrors($field, $errors) {
  683.             if (!$this->_fieldErrors) $this->_fieldErrors = array();
  684.             if (!$this->_fieldErrors[$field]) $this->_fieldErrors[$field] = array();
  685.             $this->_fieldErrors[$field] = array_merge($this->_fieldErrors[$field], $errors);
  686.         }
  687.  
  688.         /**
  689.          * Get a list of field validation errors resulting from the last validate() attempt.
  690.          * The error list returned is of the form:
  691.          *
  692.          * <code>
  693.          * array('fieldName' => array(
  694.          *       [0] => 'Error message #1',
  695.          *       [1] => 'Error message #2'
  696.          * ));
  697.          * </code>
  698.          * @return Array A list of field validation errors
  699.          */
  700.         public function getFieldErrors() {
  701.             return $this->_fieldErrors;
  702.         }
  703.  
  704.         /**
  705.          * Check if the last validation produced any model validation errors.
  706.          * @return bool TRUE if any model validators failed.
  707.          */
  708.         public function hasErrors() {
  709.             return $this->_errors && sizeof($this->_errors) > 0;
  710.         }
  711.  
  712.         /**
  713.          * Clear any model validation errors from the last validation attempt.
  714.          */
  715.         public function clearErrors() {
  716.             $this->_errors = array();
  717.         }
  718.  
  719.         /**
  720.          * Add a new model validation error.
  721.          * @param string $error The error message
  722.          */
  723.         public function addError($error) {
  724.             if (!$this->_errors) $this->_errors = array();
  725.             $this->_errors[] = $error;
  726.         }
  727.  
  728.         /**
  729.          * Get a list of model validation errors resulting from the last validate() attempt.
  730.          * @return Array A list of model validation errors
  731.          */
  732.         public function getErrors() {
  733.             return $this->_errors;
  734.         }
  735.  
  736.         /**
  737.          * Update the model's field values from a hash array of name/value pairs.
  738.          * @param Array $params A hash array of field name/value pairs
  739.          * @param Array $rawvalues A list of field names to pass through withou
  740.          *          type conversion on the values
  741.          * @param int $flags A bitmask of flags specifying information about the
  742.          *          source of the params (UPDATE_* constants)
  743.          * @see UPDATE_FROM_FORM
  744.          * @see UPDATE_FORCE_BOOLEAN
  745.          * @see UPDATE_FROM_DB
  746.          */
  747.         public function &updateModel($params, $rawvalues = null, $flags = 0) {
  748.             if ($params !== null) {
  749.                 if (($flags & UPDATE_FROM_DB) && isset($params[$this->_idField]))
  750.                     $this->_dbId = $params[$this->_idField];
  751.                 foreach ($this->_fields as $field => &$def) {
  752.                     if ($def instanceof RelationSetField)
  753.                         continue;
  754.                     if (array_key_exists($field, $params)) {
  755.                         // Don't let users update readonly fields
  756.                         if ($def->options['readonly']
  757.                                 && ($flags & UPDATE_FROM_FORM)
  758.                                 && isset($params[$this->_idField]))
  759.                             continue;
  760.                         // Blank password means don't change, since we can't access the raw password
  761.                         if ($def instanceof PasswordField && !$params[$field])
  762.                             continue;
  763.                         $v = $params[$field];
  764.                         if (!($flags & UPDATE_FROM_DB))
  765.                             $v = $def->convertToDbValue($v);
  766.                         $this->setPrimitiveFieldValue($field, $v, ($rawvalues && ($rawvalues == ALLFIELDS || in_array($field, $rawvalues))));
  767.                         // Remove found fields
  768.                         unset($params[$field]);
  769.                     } elseif ($def instanceof BooleanField && ($flags & UPDATE_FORCE_BOOLEAN)) {
  770.                         $this->setPrimitiveFieldValue($field, $def->convertToDbValue(false));
  771.                     } elseif (!array_key_exists($field, $this->_rawValues) && $field != $this->_idField) {
  772.                         $v = $def->getDefaultValue();
  773.                         if ($v instanceof ModelField)
  774.                             $v = $params[$v->getField()];
  775.                         if ($v !== null)
  776.                             $this->setPrimitiveFieldValue($field, $def->convertToDbValue($v));
  777.                     }
  778.                 }
  779.                 $this->_extraValues = array_merge($this->_extraValues, $params);
  780.                 if ($flags & UPDATE_FROM_DB)
  781.                     $this->resetChangeTracking();
  782.             } elseif (!$this->_dbId) {
  783.                 foreach ($this->_fields as $field => $def) {
  784.                     $v = $def->getDefaultValue();
  785.                     if ($v !== null)
  786.                         $this->setPrimitiveFieldValue($field, $def->convertToDbValue($v));
  787.                 }
  788.             }
  789.         }
  790.  
  791.         /**
  792.          * Persists the current model data to the data store.
  793.          *
  794.          * Validates the current model data, and attempts to save it
  795.          * to the data store, including any dependent relations that have
  796.          * been changed.
  797.          *
  798.          * @param Array $skip A list of field names to ignore during
  799.          *          cascading saves of related objects
  800.          * @throws ValidationException If data validation fails
  801.          */
  802.         public function save($skip = null) {
  803.             // Cascade-save all unsaved related fields first
  804.             if (!$skip) $skip = array();
  805.             $skip[] = $this;
  806.             foreach ($this->_fields as $field => $def) {
  807.                 if ($def instanceof ManyToOneField) {
  808.                     if (array_key_exists($field, $this->_fieldRelationValues)
  809.                             && !$this->getFieldValue($field)
  810.                             && !in_array($this->_fieldRelationValues[$field], $skip)) {
  811.                         if ($this->_fieldRelationValues[$field])
  812.                             $relid = $this->_fieldRelationValues[$field]->save($skip);
  813.                         if ($relid)
  814.                             $this->setPrimitiveFieldValue($field, $relid);
  815.                     }
  816.                 }
  817.             }
  818.  
  819.             // Don't save again if no changes were made
  820.             if ($this->_versionChanges == 0)
  821.                 return $this->_dbId;
  822.  
  823.             if ($this->validate()) {
  824.                 if ($this->_dbId) {
  825.                     // Pulled from database originally
  826.                     // (this is the only way to change the primary key)
  827.                     $this->_query->update($this, $this->_dbId);
  828.                 } elseif ($this->pk && $this->_query->filter('pk', $this->pk)->count()) {
  829.                     // If id specified, check for existence first
  830.                     $this->_dbId = $this->_query->update($this, $this->pk);
  831.                 } else {
  832.                     $this->_dbId = $this->_query->insert($this);
  833.                     $this->setFieldValue('pk', $this->_dbId);
  834.                 }
  835.                 // Save to factory cache so we don't need to reload
  836.                 $this->_query->factory->cachePut(get_class($this).':'.$this->pk, $this);
  837.                 $this->resetChangeTracking();
  838.                 return $this->_dbId;
  839.             } else {
  840.                 throw new ValidationException(get_class($this).' could not be saved (did you forget some required fields?)', $this);
  841.             }
  842.         }
  843.  
  844.         /**
