  33.     /**
  34.      * Defines a field specification for mapping to the data store.
  35.      *
  36.      * <p>Custom field types must subclass ModelField.  Model fields
  37.      * are configurated in Model::configure().</p>
  38.      *
  39.      * <p>Example:</p>
  40.      *
  41.      * <p><code>
  42.      * class Book extends Model {
  43.      *     public function configure() {
  44.      *         $this->id = new IntegerField('ID', array('pk' => true));
  45.      *         $this->title = new CharField('Title', 100, array('required' => true));
  46.      *         $this->published = new DateField('Publication Date', array(
  47.      *             'default' => CURRENT_TIMESTAMP));
  48.      *         $this->format = new CharField('Format', 20, array(
  49.      *             'options' => array('H' => 'Hardcover', 'P' => 'Paperback', 'A' => 'Audiobook')
  50.      *             ));
  51.      *         $this->setDefaultOrder('+title');
  52.      *     }
  53.      * }
  54.      * </code></p>
  55.      *
  56.      * <p>All model fields require a descriptive field name (i.e. "Publication
  57.      * Date") and an array of field options.  There are several standard
  58.      * options common to all fields, which are boolean unless otherwise
  59.      * specified:</p>
  60.      *
  61.      * <i>required</i>: The field cannot be null<br />
  62.      * <i>editable</i>: The field cannot be modified, and should
  63.      *        be hidden from display on editing interfaces<br />
  64.      * <i>readonly</i>: The field cannot be modified, but should
  65.      *        be displayed on editing interfaces<br />
  66.      * <i>blankisnull</i>: A blank string should be interpreted as
  67.      *        equivalent to NULL when saving to the data store<br />
  68.      * <i>options</i>: An array of "value" => "display name" pairs
  69.      *        specifying the valid values for this field<br />
  70.      * <i>default</i>: The default field value, if none is given<br />
  71.      * <i>unique</i>: The field's value must be unique among all
  72.      *        all models<br />
  73.      * <i>pk</i>: The field is the model's primary key.  Implies the
  74.      *        following options: required=true, editable=false,
  75.      *        blankisnull=true, unique=true
  76.      *
  77.      * @see Model
  78.      * @see Model::configure()
  79.      */

  92.         /**
  93.          * Configure a new ModelField instance.
  94.          *
  95.          * @param string $name A descriptive (human-readable) name
  96.          * @param Array $options_ An array of options
  97.          * @param Array $validators_ An array of FieldValidator objects
  98.          */
  99.         public function __construct($name, $options_ = null, $validators_ = null) {
  100.             $this->name = $name;
  101.             $this->type = get_class($this);
  102.             if ($options_ && array_key_exists('pk', $options_)) {
  103.                 $options_ = array_merge(array(
  104.                         'required' => true,
  105.                         'editable' => false,
  106.                         'blankisnull' => true,
  107.                         'blank' => false), $options_);
  108.             }
  109.             $this->options = array_merge(array('editable' => true, 'blankisnull' => true), (array)$options_);
  110.             $defaultValidators = array();
  111.             if ($options_) {
  112.                 if ($this->options['required'])
  113.                     $defaultValidators[] = array(new RequiredFieldValidator(), 'This field is required.');
  114.                 if ($length = $this->options['maxlength'])
  115.                     $defaultValidators[] = array(new MaxLengthValidator($length), 'Field value is too long.  Maximum length is '.$length.' characters.');
  116.                 if ($length = $this->options['minlength'])
  117.                     $defaultValidators[] = array(new MinLengthValidator($length), 'Field value is too short.  Minimum length is '.$length.' characters.');
  118.             }
  119.             $this->validators = array_merge($defaultValidators, (array)$validators_);
  120.         }

  122.         /**
  123.          * Add a validator to this field.
  124.          * @param FieldValidator $validator A field validator object
  125.          * @param string $message The error message to display if validation
  126.          *         fails
  127.          */
  128.         public function addValidator($validator, $message) {
  129.             $this->validators[] = array($validator, $message);
  130.         }
  131.  
  132.         /**
  133.          * Check the given value against this field's validators.
  134.          *
  135.          * @param mixed $value The field value
  136.          * @param Model $model The model instance the value belongs to
  137.          * @return bool TRUE if the field value validates
  138.          */
  139.         public function validate($value, $model) {
  140.             $this->errors = array();
  141.             $valid = true;
  142.             if (array_key_exists('options', $this->options)) {
  143.                 $validvals = array_keys($this->options['options']);
  144.                 $valueList = is_array($value) ? $value : array($value);
  145.                 foreach ($valueList as $v) {
  146.                     if ($v && !in_array($v, $validvals)) {
  147.                         $this->errors[] = 'Invalid value.';
  148.                         $valid = false;
  149.                         break;
  150.                     }
  151.                 }
  152.             }
  153.             foreach ($this->validators as $validator) {
  154.                 if (!$validator[0]->validate($value, $model)) {
  155.                     $this->errors[] = $validator[1];
  156.                     $valid = false;
  157.                 }
  158.             }
  159.             return $valid;
  160.         }
  161.  
  162.         /**
  163.          * Return a list of errors generated from the last validation.
  164.          * @return Array A list of error messages, or null if none
  165.          */
  166.         public function validationErrors() {
  167.             if (count($this->errors))
  168.                 return $this->errors;
  169.             return null;
  170.         }

  201.         /**
  202.          * Get the default value for this field.

  205.         public function getDefaultValue() {
  206.             if (isset($this->options['default']))

  208.             return null;
  209.         }

  211.         /**
  212.          * Return a human-readable description of a field value.
  213.          *
  214.          * If the value is a related object, it will display the results
  215.          * of that objects __toString() method.  If this field has an
  216.          * "options" array, if will lookup the description useing the
  217.          * given value as the key.
  218.          *
  219.          * @param mixed $value The field value
  220.          * @return string A string representation of the value
  221.          */
  222.         public function toString($value) {
  223.             if (is_object($value) && method_exists($value, '__toString'))
  224.                 return $value->__toString();
  225.             if (array_key_exists('options', $this->options))
  226.                 return strval($this->options['options'][$value]);
  227.             return strval($value);
  228.         }
  229.  
  230.         /**
  231.          * Serialize the field value to a string.
  232.          *
  233.          * This method differs from toString in that it preserves the
  234.          * value, rather than converting to a human-readable form in the
  235.          * case of an "options" mapping.
  236.          *
  237.          * @param mixed $value The field value
  238.          * @return string A string representation of the value
  239.          */
  240.         public function serialize($value) {
  241.             if (is_object($value) && method_exists($value, '__toString'))
  242.                 return $value->__toString();
  243.             return strval($value);
  244.         }
  245.  
  246.         /**
  247.          * Convert the field value to an HTML representation.
  248.          *
  249.          * Behavior is defined by the subclass.  By default, simply calls
  250.          * toString().
  251.          *
  252.          * @param mixed $value The field value
  253.          * @return string An HTML representation of the value
  254.          */
  255.         public function toHTML($value) {
  256.             return $this->toString($value);
  257.         }

  259.         public function offsetSet($index, $value) {
  260.             return false;
  261.         }
  262.         public function offsetUnset($index) {
  263.             return false;
  264.         }
  265.         public function offsetGet($index) {
  266.             if (property_exists($this, $index))
  267.                 return $this->$index;
  268.             $method = 'get'.ucfirst($index);
  269.             if (method_exists($this, $method))
  270.                 return call_user_func(array($this, $method));
  271.             return null;
  272.         }
  273.         public function offsetExists($index) {
  274.             return method_exists($this, 'get'.ucfirst($index));
  275.         }

  283.         }

  287.     /**
  288.      * Defines a field that links a model instance to an instance
  289.      * (or set of instances) of another model.
  290.      *
  291.      * Relation fields are <i>lazy-loading</i>: they only load
  292.      * related objects from the database on first access, so you
  293.      * aren't making excessive queries on objects with relation
  294.      * fields.  Lazy-loading is invisible to your code; you can
  295.      * refer to these fields as if the object is always availeble.
  296.      * See ManyToOneField for an example.
  297.      *
  298.      * @see ManyToOneField
  299.      * @see ManyToManyField
  300.      * @see OneToOneField
  301.      * @see OneToManyField
  302.      */
  303.     abstract class RelationField extends ModelField {
  304.  
  305.         public $relationName;
  306.  
  307.         protected $fldType;
  308.         protected $relModel;
  309.  
  310.         /**
  311.          * Configure a new RelationField.
  312.          *
  313.          * @param string $name_ A descriptive (human-readable) name
  314.          * @param string $relation_ The name of the Model this field is
  315.          *         linked to
  316.          * @param Array $options_ An array of options

  319.         public function __construct($name_, $relation_, $options_ = null, $validators_ = null) {
  320.             parent::__construct($name_, $options_, $validators_);
  321.             $this->relationName = $relation_;
  322.         }

  332.         abstract function setRelation($model, $value);
  333.  
  334.         /**
  335.          * Get the Model prototype that defines this field's target relation.
  336.          *
  337.          * Although this returns a Model object, it contains no useful
  338.          * field data.  It is only useful for determining metadata about
  339.          * the linked Model.
  340.          */
  341.         public function getRelationModel() {
  342.             if (!$this->relModel && $this->factory)
  343.                 $this->relModel = $this->factory->get($this->relationName)->model;
  344.             return $this->relModel;
  345.         }
  346.  
  347.         /**
  348.          * Get the field definition object (a ModelField) of the field
  349.          * of the related object that this field is linked to.
  350.          *
  351.          * This is usually the primary key of the related object.
  352.          *
  353.          * @return ModelField The field definition
  354.          */
  355.         public function getFieldType() {
  356.             if (!$this->fldType) {
  357.                 $model = $this->getRelationModel();
  358.                 $this->fldType = $model->_fields[$model->_idField];
  359.             }
  360.             return $this->fldType;
  361.         }
  362.  
  363.         /**
  364.          * @see ModelField::convertToDbValue()
  365.          */
  366.         public function convertToDbValue($value) {
  367.             $fieldType = $this->getFieldType();
  368.             if ($fieldType)
  369.                 return $fieldType->convertToDbValue($value);
  370.             return $value;
  371.         }
  372.  
  373.         /**
  374.          * @see ModelField::convertFromDbValue()
  375.          */
  376.         public function convertFromDbValue($value) {
  377.             $fieldType = $this->getFieldType();
  378.             if ($fieldType)
  379.                 return $fieldType->convertFromDbValue($value);
  380.             return $value;
  381.         }
  382.  
  383.         /**
  384.          * @see ModelField::convertValue()
  385.          */
  386.         public function convertValue($value) {
  387.             $fieldType = $this->getFieldType();
  388.             if ($fieldType)
  389.                 return $fieldType->convertValue($value);
  390.             return $value;
  391.         }

  395.     /**
  396.      * Defines a many-to-one relation between two models.
  397.      *
  398.      * In SQL terminology, this is a FOREIGN KEY field.
  399.      *
  400.      * Example:
  401.      * <code>
  402.      * class Book extends Model {
  403.      *     public function configure() {
  404.      *         $this->id = new IntegerField('ID', array('pk' => true));
  405.      *         $this->author = new OneToManyField('Author', 'Author');
  406.      *     }
  407.      * }
  408.      * 
  409.      * class Author extends Model {
  410.      *     public function configure() {
  411.      *         $this->id = new IntegerField('ID', array('pk' => true));
  412.      *         $this->name = new CharField('Name', 100);
  413.      *     }
  414.      * }
  415.      *
  416.      * // Author object is not loaded yet; we only made a single query
  417.      * $book = $bq->one();
  418.      * // Author is loaded as soon as we call the ->author field,
  419.      * // and we can access its fields like normal
  420.      * $authorname = $book->author->name;
  421.      * </code>
  422.      *
  423.      * Because relation fields are <i>lazy-loading</i>, you avoid
  424.      * excessive queries while retaining the benefit of transparent
  425.      * access to the related objects in your code.
  426.      */
  427.     class ManyToOneField extends RelationField {
  428.         
  429.         /**
  430.          * Get the Model instance that this field links to.
  431.          *
  432.          * @param Model $model The Model instance that we want to fetch
  433.          *         the related object for
  434.          * @param mixed $primitive The primary key of the related object
  435.          * @see RelationField::getRelation()
  436.          */
  437.         public function getRelation($model, $primitive = null) {
  438.             $mdef = $this->getRelationModel();
  439.             if ($primitive)
  440.                 return $mdef->getQuery()->get($primitive);
  441.             return null;
  442.         }
  443.  
  444.         /**
  445.          * Set the object that a model instance is related to, returning
  446.          * the primitive value of the primary key.
  447.          * @param Model $model The Model to set the related object for
  448.          * @param mixed $value The value to set, as either a Model object
  449.          *         or primitive primary key value
  450.          * @see RelationField::setRelation()
  451.          */
  452.         public function setRelation($model, $value) {
  453.             if ($value instanceof Model)
  454.                 return $value->pk;
  455.             return $value;
  456.         }
  457.  
  458.     }
  459.  
  460.     /**
  461.      * Creates a one-to-one relation between two models.
  462.      *
  463.      * <p>By default this functions essentially the same as a ManyToOneField,
  464.      * except it directs that the related object should also be deleted
  465.      * if the current model instance is deleted.</p>
  466.      *
  467.      * <p>OneToOneFields are unique by definition; you cannot link the target
  468.      * object to multiple source models.</p>
  469.      *
  470.      * <p>This field has several custom options:</p>
  471.      * <i>cascadeDelete</i>: delete the related object when the source object
  472.      *        is deleted.  Defaults to TRUE.<br />
  473.      * <i>reverseJoin</i>: specifies that the two models are linked by the
  474.      *        specified field (as a string) in the target model rather than the
  475.      *        source model. This then becomes a <i>synthetic field</i> that has
  476.      *        no real mapping to the database.
  477.      *
  478.      * Example:
  479.      * <code>
  480.      * class Book extends Model {
  481.      *     public function configure() {
  482.      *         $this->id = new IntegerField('ID', array('pk' => true));
  483.      *         $this->review = new OneToManyField('Review', 'Review');
  484.      *     }
  485.      * }
  486.      * 
  487.      * class Review extends Model {
  488.      *     public function configure() {
  489.      *         $this->id = new IntegerField('ID', array('pk' => true));
  490.      *         $this->rating = new IntegerField('Rating');
  491.      *     }
  492.      * }
  493.      *
  494.      * // Review object is not loaded yet; we only made a single query
  495.      * $book = $bq->one();
  496.      * // Review is loaded as soon as we call the ->review field,
  497.      * // and we can access its fields like normal
  498.      * $rating = $book->review->rating;
  499.      * </code>
  500.      */
  501.     class OneToOneField extends ManyToOneField {
  502.  
  503.         /*
  504.          * Configure a new one-to-one field.
  505.          *
  506.          * <p>This field has several custom options:</p>
  507.          * <i>cascadeDelete</i>: delete the related object when the source object
  508.          *        is deleted.  Defaults to TRUE.<br />
  509.          * <i>reverseJoin</i>: specifies that the two models are linked by the
  510.          *        specified field (as a string) in the target model rather than the
  511.          *        source model. This then becomes a <i>synthetic field</i> that has
  512.          *        no real mapping to the database.
  513.          *
  514.          * @param string $name_ A descriptive (human-readable) name
  515.          * @param string $relation_ The name of the Model this field is
  516.          *        linked to
  517.          * @param Array $options_ An array of options
  518.          * @param Array $validators_ An array of FieldValidator objects
  519.          */
  520.         public function __construct($name_, $relation_, $options_ = null, $validators_ = null) {
  521.             parent::__construct($name_, $relation_, $options_, $validators_);
  522.             $this->options['unique'] = true;
  523.             $this->options['cascadeDelete'] = true;
  524.         }
  525.  
  526.         /**
  527.          * @see ManyToOneField::getRelation()
  528.          */
  529.         public function getRelation($model, $primitive = null) {
  530.             if ($this->options['reverseJoin']) {
  531.                 $mdef = $this->getRelationModel();
  532.                 if ($model)
  533.                     return $mdef->getQuery()->filter(
  534.                         $this->options['reverseJoin'], $model->pk)->any();
  535.                 return null;
  536.             } else
  537.                 return parent::getRelation($model, $primitive);
  538.         }
  539.  
  540.         /**
  541.          * @see ManyToOneField::setRelation()
  542.          */
  543.         public function setRelation($model, $value) {
  544.             if ($this->options['reverseJoin']) {
  545.                 if ($value instanceof Model) {
  546.                     $value[$this->options['reverseJoin']] = $model->pk;
  547.                     return $value->pk;
  548.                 } else
  549.                     return $value;
  550.             }
  551.             return parent::setRelation($model, $value);
  552.         }
  553.  
  554.     }
  555.  
  556.     /**
  557.      * A relation field that returns a set of related models, rather
  558.      * than the single model instance that ManyToOneField and OneToOneField
  559.      * return.
  560.      *
  561.      * <p>This field has a custom option:</p>
  562.      * <i>joinField</i>: The name of the field on the target model that
  563.      *        links back to this model.  Similar to "reverseJoin" for
  564.      *        OneToOneField.
  565.      */
  566.     abstract class RelationSetField extends RelationField {
  567.  
  568.         protected $joinField;
  569.  
  570.         /*
  571.          * Configure a new one-to-one field.
  572.          *
  573.          * <p>This field has a custom option:</p>
  574.          * <i>joinField</i>: The name of the field on the target model that
  575.          *        links back to this model.  Similar to "reverseJoin" for
  576.          *        OneToOneField.
  577.          *
  578.          * @param string $name_ A descriptive (human-readable) name
  579.          * @param string $relation_ The name of the Model this field is
  580.          *        linked to
  581.          * @param Array $options_ An array of options
  582.          * @param Array $validators_ An array of FieldValidator objects
  583.          */
  584.         public function __construct($name_, $relation_, $options_ = null, $validators_ = null) {
  585.             parent::__construct($name_, $relation_, $options_, $validators_);
  586.             $this->joinField = $this->options['joinField'];
  587.         }
  588.         
  589.         /**
  590.          * Getter to provide access to RelationSetField::$joinField that
  591.          * lazy-loads the target model definition on first access.
  592.          */
  593.         public function __get($name) {
  594.             if ($name == 'joinField' && !$this->joinField)
  595.                 $this->joinField = strtolower(get_class($this->query->model));
  596.             return $this->$name;
  597.         }
  598.  
  599.         /**
  600.          * @see RelationField::setRelation()
  601.          */
  602.         public function setRelation($model, $value) {
  603.             $set = $this->getRelation($model);
  604.             $set->set($value);
  605.         }
  606.  
  607.     }
  608.  
  609.     /**
  610.      * A field that returns a set of models related to this one, by
  611.      * a join field on the target model that points to this model's
  612.      * primary key.
  613.      *
  614.      * <p>This field has a custom option:</p>
  615.      * <i>joinField</i>: The name of the field on the target model that
  616.      *        links back to this model.  Defaults to the source model's
  617.      *        name, lower-cased.
  618.      *
  619.      * Example:
  620.      * <code>
  621.      * class Author extends Model {
  622.      *     public function configure() {
  623.      *         $this->id = new IntegerField('ID', array('pk' => true));
  624.      *         $this->books = new OneToManyField('Books', 'Book'):
  625.      *     }
  626.      * }
  627.      *
  628.      * class Book extends Model {
  629.      *     public function configure() {
  630.      *         $this->id = new IntegerField('ID', array('pk' => true));
  631.      *         $this->title = new CharField('Title', 100);
  632.      *     }
  633.      * }
  634.      * 
  635.      * // Book list is not loaded yet
  636.      * $author = $aq->one();
  637.      * // Book list is loaded as soon as we call the ->books field,
  638.      * foreach ($author->books as $book) {
  639.      *     echo $book->title;
  640.      * }
  641.      * </code>
  642.      *
  643.      * @see RelationSetField
  644.      * @see RelationSet
  645.      */
  646.     class OneToManyField extends RelationSetField {
  647.  
  648.         /**
  649.          * Get a list of model instances that are related to this model
  650.          * instance.
  651.          *
  652.          * @param Model $model The model instance to fetch relations for
  653.          * @param mixed $primitive The key value the related objects
  654.          *         are linked to (usually the current model's primary key)
  655.          * @return RelationSet An iterable RelationSet object
  656.          * @see RelationSet
  657.          */
  658.         public function getRelation($model, $primitive = null) {
  659.             return new OneToManyRelationSet($model, $this);
  660.         }
  661.  
  662.     }
  663.  
  664.     /**
  665.      * A field that returns a set of models related to this one, using
  666.      * a third intermediary Model object to create a many-to-many mapping.
  667.      *
  668.      * <p>This field has several custom options:</p>
  669.      * <i>joinField</i>: The name of the field on the intermediary model that
  670.      *        links to the source model's primary key.  Defaults to the source
  671.      *        model's name, lower-cased.
  672.      * <i>targetField</i>: The name of the field on the intermediary model that
  673.      *        links to the target model's primary key.  Defaults to the target
  674.      *        model's name, lower-cased.
  675.      *
  676.      * Example:
  677.      * <code>
  678.      * class Author extends Model {
  679.      *     public function configure() {
  680.      *         $this->id = new IntegerField('ID', array('pk' => true));
  681.      *         $this->genres = new ManyToManyField('Genres', 'Genre', 'AuthorGenre'):
  682.      *     }
  683.      * }
  684.      *
  685.      * class AuthorGenre extends Model {
  686.      *     public function configure() {
  687.      *         $this->author = new ManyToOneField('Author');
  688.      *         $this->genre = new ManyToOneField('Genre');
  689.      *     }
  690.      * }
  691.      * class Genre extends Model {
  692.      *     public function configure() {
  693.      *         $this->id = new IntegerField('ID', array('pk' => true));
  694.      *         $this->name = new CharField('Name', 100);
  695.      *     }
  696.      * }
  697.      * 
  698.      * // Genre list is not loaded yet
  699.      * $author = $aq->one();
  700.      * // Genre list is loaded as soon as we call the ->genres field,
  701.      * foreach ($author->genres as $genre) {
  702.      *     echo $genre->name;
  703.      * }
  704.      *
  705.      * // Now let's add a new genre to an author
  706.      * $genre = $gq->filter('name', 'Science Fiction')->one();
  707.      * $author->genres->add($genre);
  708.      * $author->save();
  709.      * </code>
  710.      *
  711.      * Although you need the intermediary field definition to manage the
  712.      * relationships, you should usually not need to access it directly,
  713.      * unless you need to store additional information about the relation.
  714.      * ModelQuery adds and removes records of that model as needed to
  715.      * maintain the relationship links.
  716.      *
  717.      * @see RelationSetField
  718.      * @see RelationSet
  719.      */
  720.     class ManyToManyField extends RelationSetField {
  721.  
  722.         public $joinModel;
  723.         public $targetField;
  724.  
  725.         protected $joinModelObj;
  726.  
  727.         /*
  728.          * Configure a new many-to-many field.
  729.          *
  730.          * <p>This field has several custom options:</p>
  731.          * <i>joinField</i>: The name of the field on the intermediary model that
  732.          *        links to the source model's primary key.  Defaults to the source
  733.          *        model's name, lower-cased.
  734.          * <i>targetField</i>: The name of the field on the intermediary model that
  735.          *        links to the target model's primary key.  Defaults to the target
  736.          *        model's name, lower-cased.
  737.          *
  738.          * @param string $name_ A descriptive (human-readable) name
  739.          * @param string $relation_ The name of the Model this field is
  740.          *        linked to
  741.          * @param string $joinModel_ The name of the intermediary model that
  742.          *        links the source and target models
  743.          * @param Array $options_ An array of options
  744.          * @param Array $validators_ An array of FieldValidator objects
  745.          */
  746.         public function __construct($name_, $relation_, $joinModel_, $options_ = null, $validators_ = null) {
  747.             parent::__construct($name_, $relation_, $options_, $validators_);
  748.             $this->joinModel = $joinModel_;
  749.             $this->targetField = $this->options['targetField']
  750.                 ? $this->options['targetField']
  751.                 : strtolower($relation_);
  752.         }
  753.  
  754.         /**
  755.          * Get the Model prototype object for the intermediary joining model.
  756.          */
  757.         public function getJoinModel() {
  758.             if (!$this->joinModelObj)
  759.                 $this->joinModelObj = $this->factory->get($this->joinModel)->model;
  760.             return $this->joinModelObj;
  761.         }
  762.  
  763.         /**
  764.          * Get a list of model instances that are related to this model
  765.          * instance.
  766.          *
  767.          * @param Model $model The model instance to fetch relations for
  768.          * @param mixed $primitive The key value the related objects
  769.          *         are linked to (usually the current model's primary key)
  770.          * @return RelationSet An iterable RelationSet object
  771.          * @see RelationSet
  772.          */
  773.         public function getRelation($model, $primitive = null) {
  774.             return new ManyToManyRelationSet($model, $this);
  775.         }
  776.  
  777.     }

  779.     /**
  780.      * Field contains an integer.
  781.      */
  782.     class IntegerField extends ModelField {
  783.  
  784.         /**
  785.          * @see ModelField::__construct()
  786.          */
  787.         public function __construct($name_, $options_ = null, $validators_ = null) {

  789.             $this->validators[] = array(new NumericValidator(), 'Invalid number format.');

  791.  
  792.         public function convertValue($value) {
  793.             if ($value === '' || $value === null)

  796.         }
  797.  
  798.     }
  799.  
  800.     /**
  801.      * Field contains an ordered integer.
  802.      *
  803.      * OrderedField fields get special handling, because ModelQuery needs
  804.      * to manage the ordering to ensure there are no gaps or duplicates
  805.      * in the data store.
  806.      *
  807.      * Example:
  808.      * <code>
  809.      * class Portfolio extends Model {
  810.      *     public function configure() {
  811.      *         $this->id = new IntegerField('ID', array('pk' => true));
  812.      *         $this->user = new ManyToOneField('User', 'User');
  813.      *         $this->sort_order = new OrderedField('Sort Order', array('user'));
  814.      *         $this->name = new CharField('ID', 100);
  815.      *     }
  816.      * }
  817.      * 
  818.      * // Start with a user
  819.      * $user1 = $uq->get(1);
  820.      *
  821.      * // Create a few portofolios.  As we add them, ModelQuery assigns
  822.      * // the sort_order field automatically.
  823.      * $up1 = $pq->create(array('user' => $user1, 'title' = 'U1P1'));
  824.      * $up1->save();
  825.      * // $up1->sort_order is now "1"
  826.      * $up2 = $pq->create(array('user' => $user1, 'title' = 'U1P2'));
  827.      * $up2->save();
  828.      * // $up2->sort_order is now "2"
  829.      * $up3 = $pq->create(array('user' => $user1, 'title' = 'U1P3'));
  830.      * $up3->save();
  831.      * // $up3->sort_order is now "3"
