  288.     class PasswordValidator implements FieldValidator {
  289.  
  290.         private $numrequired;
  291.         private $symrequired;
  292.         private $charpattern;
  293.  
  294.         /** 
  295.          * Create a new validator.
  296.          * @param bool $numrequired Whether a numeric character is required
  297.          *         in the password
  298.          * @param bool $symrequired Whether a non-alphanumeric character is
  299.          *         required in the password
  300.          * @param string $charpattern A regular expression to test the password
  301.          *         against (advanced usage)
  302.          */
  303.         public function __construct($numrequired = false, $symrequired = false, $charpattern = null) {
  304.             $this->numrequired = $numrequired;
  305.             $this->symrequired = $symrequired;
  306.             $this->charpattern = $charpattern;
  307.         }
  308.  
  309.         public function validate($value, $model) {
  310.             if (!$value)
  311.                 return false;
  312.  
  313.             // Only validate non-hashed strings (<32 chars). Hackish.
  314.             if ($value && strlen($value) == 32 && preg_match('/^[A-Za-z0-9]*$/', $value))
  315.                 return true;
  316.  
  317.             $hassym = false;
  318.             $hasnum = false;
  319.  
  320.             if ($this->charpattern && !preg_match('/^'.$this->charpattern.'*$/', $value))
  321.                 return false;
  322.  
  323.             for ($i = 0; $i < strlen($value); ++$i) {
  324.                 $o = ord($value{$i});
  325.                 if ($o >= 48 && $o <= 57)
  326.                     $hasnum = true;
  327.                 elseif (($o >= 32 && $o <= 47)
  328.                         || ($o >= 58 && $o <= 64)
  329.                         || ($o >= 91 && $o <= 96)
  330.                         || ($o >= 123 && $o <= 126))
  331.                     $hassym = true;
  332.             }
  333.  
  334.             if (($this->numrequired && !$hasnum) || ($this->symrequired && !$hassym))
  335.                 return false;
  336.  
  337.             return true;
  338.         }
  339.     }

  385.             for ($i = 0; $i < sizeof($local_array); $i++) {

  387.                     return false;
  388.                 }
  389.             }  
  390.             if (!ereg("^\[?[0-9\.]+\]?$", $email_array[1])) { // Check if domain is IP. If not, it should be valid domain name

  394.                 }

  396.                     if (!ereg("^(([A-Za-z0-9][A-Za-z0-9-]{0,61}[A-Za-z0-9])|([A-Za-z0-9]+))$", $domain_array[$i])) {
  397.                         return false;
  398.                     }
  399.                 }
  400.             }
  401.             return true;
  402.         }
