  33.     /**
  34.      * Core class for query creation in ModelQuery.
  35.      *
  36.      * All queries constructed with ModelQuery use a chain of one or
  37.      * more QueryFilter objects.
  38.      * 
  39.      * <b>Basic Usage Examples</b>
  40.      * <code>
  41.      * // Get the User QueryHandler object (which is also a QueryFilter) from our QueryFactory
  42.      * $uq = $qf->get('User');
  43.      * 
  44.      * // Get all users
  45.      * $users = $uq->all();
  46.      * 
  47.      * // Get all users, sorted by username
  48.      * $users = $uq->order('username');
  49.      * 
  50.      * // Get all verified users
  51.      * $users = $uq->filter('verified', true);
  52.      * 
  53.      * // Delete all unverified users created before Jan 1
  54.      * $uq->filter('verified', false, 'created:lte', mktime(0,0,0,1,1,2008))->delete();
  55.      * 
  56.      * // Get an array of emails for all unverified users with non-null email addresses
  57.      * $values = $uq->filter('verified', true, 'email:isnull', false)->values('email');
  58.      * 
  59.      * // Get a list of all users with gmail.com email addresses
  60.      * $users = $uq->filter('email:endswith', '@gmail.com');
  61.      * 
  62.      * // Get all unverified users except those in group 5
  63.      * $values = $uq->filter('verified', false)->exclude('group', 5);
  64.      * 
  65.      * // Set all users in group 5 to verified
  66.      * $uq->filter('group', 5)->update(array('verified', true));
  67.      * 
  68.      * // Get the current user count
  69.      * $count = $uq->count();
  70.      * </code>
  71.      * 
  72.      * <b>Filter Chaining</b>
  73.      * 
  74.      * You will notice that except for methods that return immediate
  75.      * query results (hash(), select(), raw(), delete(), get(), etc),
  76.      * most filtering methods in this class (filter(), exclude(),
  77.      * slice(), order(), etc) return a new QueryFilter object that
  78.      * chains to the one it was called from, without altering its
  79.      * original state or executing a query against the database.
  80.      * Chaining filters in this way allows you to branch in several
  81.      * different directions with a query without having to start
  82.      * over again.
  83.      * 
  84.      * <code>
  85.      * // Base query is limited to only users in group 5
  86.      * $query = $uq->filter('group', 5);
  87.      * 
  88.      * // Get total number of users in group 5
  89.      * $total = $query->count();
  90.      * 
  91.      * // ...but only load the first 10
  92.      * $users = $query->slice(10);
  93.      * </code>
  94.      * 
  95.      * <b>Accessing the Result Set</b>
  96.      * 
  97.      * As mentioned above, most of the filter methods return a QueryFilter object,
  98.      * but no query is actually executed against the database yet. So how do we get
  99.      * the actual query results?
  100.      * 
  101.      * Just start accessing it as a regular PHP array. As soon as you call it
  102.      * in an array context (either as part of a foreach loop, or by trying to directly
  103.      * access an array index on it), the underlying query is executed on the database
  104.      * and the result set is created.
  105.      * 
  106.      * <code>
  107.      * // Filter applied, but no query execution yet
  108.      * $users = $uq->filter('group', 5);
  109.      * 
  110.      * // foreach loop forces query execution
  111.      * foreach ($users as $user) {
  112.      * 
  113.      *     // Like QueryFilter, Model instances can also be accessed as objects or arrays:
  114.      *     echo $user->username;
  115.      *     // ... is equivalent to:
  116.      *     echo $user['username'];
  117.      * 
  118.      * }
  119.      * </code>
  120.      * 
  121.      * <b>Cursors</b>
  122.      * 
  123.      * If you have too many rows to load in memory, you can also use a database
  124.      * cursor:
  125.      * 
  126.      * <code>
  127.      * // Get a CursorIterator object for this query
  128.      * $users = $uq->filter('group', 5)->cursor();
  129.      * 
  130.      * // CursorIterator can also be used normally as an array, but it keeps the database
  131.      * // connection open and loads rows as needed:
  132.      * foreach ($users as $user) {
  133.      *     // Do something
  134.      * }
  135.      * 
  136.      * // Manually close down the cursor
  137.      * $users->close();
  138.      * </code>
  139.      * 
  140.      * @package modelquery
  141.      * @see QueryHandler
  142.      * @see CursorIterator

  218.         /**
  219.          * Filter models by a field value.
  220.          * 
  221.          * <p>This method takes an arbitrary number of parameters, but they
  222.          * must come in pairs, such as:</p>
  223.          *
  224.          * <p><code>
  225.          * $uq->filter("username:exact", "jjongsma")
  226.          * $uq->filter("verified", true, "created:gte", mktime(0,0,0,1,1,2008));
  227.          * </code></p>
  228.          * 
  229.          * <p>The first parameter of a pair consists of a field name, and an
  230.          * optional modifier separated by a colon (:).  The second parameter
  231.          * of a pair is the field value to filter by.  The type of the second
  232.          * parameter depends on the field type and the modifier used.  For
  233.          * example, for a date field, you would pass in a numeric timestamp.
  234.          * For an :in modifier (see below), you would pass in an array of values
  235.          * as the second parameter.  If no modifier is specified, "exact" is
  236.          * assumed.</p>
  237.          * 
  238.          * <p>If multiple pairs of parameters are given, records must match
  239.          * ALL of the filters specified.</p>
  240.          * 
  241.          * <p><b>Available modifiers:</b><br />
  242.          * <i>exact</i>: Match records exactly against the given field value<br />
  243.          * <i>iexact</i>: Match records exactly against the given field value, case-insensitive<br />
  244.          * <i>contains</i>: Match records that contain the given value anywhere in the field<br />
  245.          * <i>icontains</i>: Match records that contain the given value anywhere in the field, case-insensitive<br />
  246.          * <i>gt</i>: Match records that have a value greater than the given value<br />
  247.          * <i>gte</i>: Match records that have a value greater than or equal to the given value<br />
  248.          * <i>lt</i>: Match records that have a value less than the given value<br />
  249.          * <i>lte</i>: Match records that have a value less than or equal to the given value<br />
  250.          * <i>in</i>: Match records that have a value that matches any value in the given array<br />
  251.          * <i>startswith</i>: Match records with a value that starts with the given string<br />
  252.          * <i>istartswith</i>: Match records with a value that starts with the given string, case-insensitive<br />
  253.          * <i>endswith</i>: Match records with a value that ends with the given string<br />
  254.          * <i>iendswith</i>: Match records with a value that ends with the given string, case-insensitive<br />
  255.          * <i>range</i>: Match records within the specified range (given as a 2-item array), inclusive<br />
  256.          * <i>isnull</i>: Match records with null values (if parameter is TRUE) or non-null values (if FALSE)</p>
  257.          * 
  258.          * <p>In addition to filtering by fields that are part of the original model,
  259.          * you can also filter by related objects.  For example, if the User model
  260.          * has a ManyToOneField called "preferences", you could filter on:</p>
  261.          * 
  262.          * <code>
  263.          * $uq->filter('preferences.homepage', '/news/finance.php');
  264.          * </code>
  265.          * 
  266.          * <p>This would filter on the "homepage" field belonging to the Preferences model
  267.          * linked to each User record.  You can follow relations as far as they go using
  268.          * this dot notation, not just a single level; QueryFilter will add all the
  269.          * necessary joins.</p>
  270.          * 
  271.          * @return QueryFilter The new QueryFilter at the end of the filter chain
  272.          */

  331.         /** Parse function argument list to split into field/value filter pairs. */
  332.         private function getFilters($args) {
  333.             if (sizeof($args) == 1 && is_array($args[0])) {
  334.                 $filters = array();
  335.                 $keys = array_keys($args[0]);
  336.                 foreach ($keys as $key) {
  337.                     $filters[] = $key;
  338.                     $filters[] = $args[0][$key];
  339.                 }
  340.                 return $filters;
  341.             } else
  342.                 return $args;

  345.         /**
  346.          * Adds extra fields from related tables to the results
  347.          * of this query, to reduce additional queries.
  348.          *
  349.          * The parameter is of the form "reference_name" => "field_name",
  350.          * where reference_name is the name of the field as it will be
  351.          * named in the query result object, and field_name is the
  352.          * original field name (or path, if you are pulling in a field
  353.          * from a related object).
  354.          *
  355.          * <code>
  356.          * $user = $uq->extra(array('homepage' => 'preferences.homepage'));
  357.          * echo $user->homepage:
  358.          * </code>
  359.          *
  360.          * While the original user object did not have a "homepage"
  361.          * field, the query results now include the value of that
  362.          * field from the related "preferences" object.  This value
  363.          * could also be retrieved with $user->preferences->homepage,
  364.          * but it would require an additional database query since
  365.          * the entire preferences object would be loaded on demand.
  366.          *
  367.          * @param Array $select A hash array of additional fields to include.
  368.          * @param string $where <i>Deprecated, use {@see QueryFilter::condition()}</i>.
  369.          * @param Array $params <i>Deprecated, use {@see QueryFilter::condition()}</i>.
  370.          * @return QueryFilter The new QueryFilter at the end of the filter chain
  371.          */
  372.         public function &extra($select = null, $where = null, $params = null) {

  722.          * Get an aggregate field value and return the results immediately.
  723.          *
  724.          * The following two statements are equivalent:
  725.          *
  726.          * <code>
  727.          * $query->addAggregate('id', 'id', 'MAX')->one()->id;
  728.          * </code>
  729.          * and:
  730.          * <code>
  731.          * $query->returnAggregate('id', 'id', 'MAX', null, true);
  732.          * </code>
  733.          *
  734.          * This only returns a single value, so if you are specifying
  735.          * fields to group by, make sure you are also filtering to
  736.          * get the correct group, or you will only get the first
  737.          * value of a result set.
  738.          *
  739.          * @param string $name The new field name
  740.          * @param string $field The field name to aggregate
  741.          * @param string $fn The SQL aggregate function (MIN, MAX, etc)
  742.          * @param Array $groupby A list of fields to group results by
  743.          * @param bool $convert Whether to type-convert the resulting value.
  744.          *             For example, if you are computing an average of integer
  745.          *             fields, you probably don't want to convert the resulting
  746.          *             float value to an integer.
  747.          * @return mixed The aggregate field value
  748.          */
  749.         public function returnAggregate($name, $field, $fn, $groupby = null, $convert = false) {
  750.             $filter = new QueryFilter($this->queryhandler, $this);
  751.             list($name, $fieldObj) = $filter->addAggregate($name, $field, $fn, $groupby);
  752.             $result = $filter->raw();
  753.             $aggregate = $result[0][$name];
  754.             if ($convert) {
  755.                 return $fieldObj->convertFromDbValue($aggregate);
  756.             } else {
  757.                 return $aggregate;
  758.             }
  759.         }

  767.          * Applies a filter to the current filter, rather than creating a new
  768.          * QueryFilter and adding to the end of the chain.
