  69.         /**
  70.          * Get a configured QueryHandler object by name.
  71.          * <code>$qf->get('User')</code> will look for a file named User.class.php
  72.          * in the list of model directories specified in the constructor, and upon
  73.          * finding one will include it and attempt to instantiate a <b>User</b>
  74.          * class.
  75.          *
  76.          * @param string $name The model name.
  77.          * @return QueryHandler A QueryHandler object configured for querying the
  78.          *          specified model type.
  79.          */
  80.         public function &get($name) {
  81.             if (!array_key_exists($name, $this->objectCache)) {
  82.                 $found = false;
  83.                 foreach ($this->modelRoots as $modelDir) {
  84.                     $includeFile = $modelDir.'/'.$name.'.class.php';
  85.                     if (QueryFactory::file_exists_in_path($includeFile)) {
  86.                         require_once($includeFile);
  87.                         $found = true;
  88.                         break;
  89.                     }
  90.                 }
  91.                 if ($found) {
  92.                     $mq = new QueryHandler($name, $this, $this->appName);
  93.                     $this->objectCache[$name] =& $mq;
  94.                 }
  95.             }
  96.             return $this->objectCache[$name];
  97.         }

  99.         /**
  100.          * Default property getter.
  101.           * 
  102.          * Allows alternate syntax for fetching a QueryHandler object.  The calls
  103.          * $qf->User and $qf->get('User') are equivalent.
  104.         */

  106.             return $this->get($name);
  107.         }

  110.          * Configure QueryHandler objects for all model objects found in the

  112.          *
  113.          * Because instantiating large numbers of QueryHandler objects can be

  118.             $classes = QueryFactory::getModelClasses($this->modelRoots);
  119.             foreach ($classes as $c => $f)
  120.                 $this->get($c);
  121.         }

  140.         /**
  141.          * Begin a transaction.
  142.          * @exception TransactionException if a transaction is already in progress
  143.          */
  144.         public function begin() {
  145.             if ($this->transOpen)
  146.                 throw new TransactionException('A transaction is already in progress.');
  147.             $this->transOpen = true;
  148.             $this->transFailed = false;
  149.             $this->conn->BeginTrans();
  150.         }
  151.  
  152.         /**
  153.          * Commit the current transaction to the database.
  154.          * @exception TransactionException if no transaction is in progress
  155.          */
  156.         public function commit() {
  157.             if (!$this->transOpen)
  158.                 throw new TransactionException('No transaction is in progress.');
  159.             $this->conn->CommitTrans();
  160.             $this->transOpen = false;
  161.         }
  162.  
  163.         /**
  164.          * Rollback the current transaction.
  165.          * @exception TransactionException if no transaction is in progress
  166.          */
  167.         public function rollback() {
  168.             if (!$this->transOpen)
  169.                 throw new TransactionException('No transaction is in progress.');
  170.             $this->conn->RollbackTrans();
  171.             $this->transOpen = false;
  172.         }

  174.         /**
  175.          * Cause any currently open transaction to fail, triggering a rollback.

  177.         public function fail() {
  178.             $this->transFailed = true;
  179.         }
  180.  
  181.         /**
  182.          * Complete a transaction,
  183.          *
  184.          * If the transaction was manually cancelled with fail(), this calls
  185.          * rollback().  Otherwise, commit() is called.

  187.         public function complete() {
  188.             if ($this->transFailed)
  189.                 $this->rollback();
  190.             else
  191.                 $this->commit();
  192.         }

  254.         /**
  255.          * Check if a file exists in any directory of PHP's include_path.
  256.          * @param string $file The file/relative path to search for
  257.          * @return TRUE if the file is found and able to be included
  258.          */
  259.         private static function file_exists_in_path($file) {
  260.             if (file_exists($file))
  261.                 return TRUE;
  262.             if (!QueryFactory::$path)
  263.                 QueryFactory::$path = explode(':', ini_get('include_path'));
  264.             foreach(QueryFactory::$path as $dir)
  265.                 if (file_exists($dir.'/'.$file))
  266.                     return TRUE;
  267.             return FALSE;
  268.         }
  269.  
  270.         /**
  271.          * Resolve the path of a file in any directory of PHP's include_path.
  272.          * @param string $file The file/relative path to search for
  273.          * @return string The full path of the file, if found
  274.          */
  275.         private static function find_file_in_path($file) {
  276.             if (file_exists($file))
  277.                 return TRUE;
  278.             if (!QueryFactory::$path)
  279.                 QueryFactory::$path = explode(':', ini_get('include_path'));
  280.             foreach(QueryFactory::$path as $dir)
  281.                 if (file_exists($dir.'/'.$file))
  282.                     return $dir.'/'.$file;
  283.             return null;
  284.         }

  286.         /**
  287.          * Load the class definitions of all known models.
  288.          *
  289.          * Searches the specfied list of directories for model definitions,
  290.          * and loads the class files without instantiating the model classes.
  291.          * This can be useful in combination with precacheModels().  After
  292.          * calling precacheModels(), a QueryFactory can be serialized to a
  293.          * shared cache (like memcache or xcache), and on future requests,
  294.          * calling preloadModelClasses() before deserialization avoids
  295.          * any class loading errors.
  296.          *
  297.          * @param Array $roots A list of directories to search for models in
  298.         */
  299.         public static function preloadModelClasses($roots) {
  300.             $classes = QueryFactory::getModelClasses($roots);
  301.             foreach ($classes as $c => $f)
  302.                 require_once($f);
  303.         }
  304.  
  305.         /**
  306.          * Get the class names of all model definitions found in the search path.
  307.          *
  308.          * Any class names returned will have already had their class definitions
  309.          * loaded into memory, and are ready for instantiation.
  310.          */
  311.         private static function getModelClasses($roots) {
  312.             $classes = array();
  313.             foreach ($roots as $root) {
  314.                 $dir = QueryFactory::find_file_in_path($root);
  315.                 if (is_dir($dir)) {
  316.                     $d = opendir($dir);
  317.                     while ($f = readdir($d))
  318.                         if (substr($f, -10) == '.class.php')
  319.                             $classes[substr($f, 0, -10)] = $dir.'/'.$f;
  320.                     closedir($d);
  321.                 }
  322.             }
  323.             return $classes;
  324.         }
